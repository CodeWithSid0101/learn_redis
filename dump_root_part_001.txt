import os
from urllib.parse import quote_plus
from sqlalchemy import create_engine, inspect
from sqlalchemy.exc import OperationalError
from sqlalchemy.orm import sessionmaker, declarative_base
from fastapi import HTTPException, status

# Password is read from an environment variable and URL-encoded so special
# characters (like '@' in "Sidd@200428") don't break the connection string
# parsing — without encoding, SQLAlchemy misreads the second '@' as the
# separator before the host, producing a garbage hostname.
#   Windows (CMD): set MYSQL_ROOT_PASSWORD=Sidd@200428
DB_PASSWORD_RAW = os.environ.get("MYSQL_ROOT_PASSWORD", "")
DB_PASSWORD = quote_plus(DB_PASSWORD_RAW)

SQL_URL = f"mysql+pymysql://root:{DB_PASSWORD}@127.0.0.1:3306/local_operations"

engine = create_engine(SQL_URL, pool_pre_ping=True)
SessionLocal = sessionmaker(bind=engine, autocommit=False, autoflush=False)
Base = declarative_base()

TABLE_NAME = "sample"


def get_db():
    try:
        inspector = inspect(engine)
        table_exists = inspector.has_table(TABLE_NAME)
    except OperationalError as e:
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail=f"Could not connect to the database: {e}",
        )

    if not table_exists:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail=f"No table found named '{TABLE_NAME}'. Please create it before using this API.",
        )

    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

import io
import time
from fastapi import FastAPI, HTTPException, status, Depends, UploadFile, File
from sqlalchemy.orm import Session

import models
from database import get_db
from schemas import OrderCreate, OrderFetchRequest, OrderResponse
from redis_client import (
    get_cache, set_cache, delete_cache,
    cache_key_for_order, cache_key_for_list, invalidate_list_cache,
)
from seed_data import seed_from_excel, validate_excel_headers

app = FastAPI(title="Orders API with Redis Caching")


@app.get("/orders/status", status_code=status.HTTP_200_OK)
def orders_status(db: Session = Depends(get_db)):
    count = db.query(models.Order).count()
    return {"total_orders_in_db": count, "seeded": count > 0}


@app.post("/orders/upload-excel", status_code=status.HTTP_200_OK)
async def upload_orders_excel(file: UploadFile = File(...), db: Session = Depends(get_db)):
    filename = (file.filename or "").lower()
    if not filename.endswith((".xlsx", ".xlsm", ".xls")):
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail="Only .xlsx, .xlsm, or .xls files are supported.")

    contents = await file.read()
    if not contents:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail="Uploaded file is empty.")

    file_stream = io.BytesIO(contents)
    is_valid, message = validate_excel_headers(file_stream)
    if not is_valid:
        raise HTTPException(status_code=status.HTTP_422_UNPROCESSABLE_ENTITY, detail=f"Excel validation failed: {message}")

    file_stream.seek(0)
    summary = seed_from_excel(db, excel_path=file_stream)
    invalidate_list_cache()
    return {"detail": "Excel validated and orders uploaded successfully.", **summary}


@app.post("/orders/seed", status_code=status.HTTP_200_OK)
def seed_orders_from_excel(db: Session = Depends(get_db)):
    summary = seed_from_excel(db)
    invalidate_list_cache()
    return {"detail": "Seed complete", **summary}


@app.post("/orders/create", response_model=OrderResponse, status_code=status.HTTP_201_CREATED)
def create_order(payload: OrderCreate, db: Session = Depends(get_db)):
    start = time.perf_counter()

    existing = db.query(models.Order).filter(models.Order.order_id == payload.order_id).first()
    if existing:
        raise HTTPException(status_code=status.HTTP_409_CONFLICT, detail="order_id already exists")

    order = models.Order(**payload.model_dump())
    db.add(order)
    db.commit()
    db.refresh(order)

    response_data = OrderResponse.model_validate(order).model_dump()
    response_data["cache_hit"] = False
    response_data["response_time_ms"] = round((time.perf_counter() - start) * 1000, 2)

    set_cache(cache_key_for_order(order.order_id), response_data)
    invalidate_list_cache()

    return response_data


@app.post("/orders/fetch")
def fetch_orders(payload: OrderFetchRequest, db: Session = Depends(get_db)):
    start = time.perf_counter()

    if payload.order_id:
        key = cache_key_for_order(payload.order_id)
        cached = get_cache(key)
        if cached is not None:
            cached["cache_hit"] = True
            cached["response_time_ms"] = round((time.perf_counter() - start) * 1000, 2)
            return cached

        order = db.query(models.Order).filter(models.Order.order_id == payload.order_id).first()
        if not order:
            raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Order not found")

        response_data = OrderResponse.model_validate(order).model_dump()
        response_data["cache_hit"] = False
        response_data["response_time_ms"] = round((time.perf_counter() - start) * 1000, 2)
        set_cache(key, response_data)
        return response_data

    key = cache_key_for_list()
    cached = get_cache(key)
    if cached is not None:
        for row in cached:
            row["cache_hit"] = True
            row["response_time_ms"] = round((time.perf_counter() - start) * 1000, 2)
        return cached

    orders = db.query(models.Order).all()
    response_data = [OrderResponse.model_validate(o).model_dump() for o in orders]
    for row in response_data:
        row["cache_hit"] = False
        row["response_time_ms"] = round((time.perf_counter() - start) * 1000, 2)
    set_cache(key, response_data)
    return response_data


@app.post("/orders/update", response_model=OrderResponse)
def update_order(order_id: str, payload: OrderCreate, db: Session = Depends(get_db)):
    start = time.perf_counter()
    order = db.query(models.Order).filter(models.Order.order_id == order_id).first()
    if not order:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Order not found")

    for field, value in payload.model_dump().items():
        setattr(order, field, value)
    db.commit()
    db.refresh(order)

    response_data = OrderResponse.model_validate(order).model_dump()
    response_data["cache_hit"] = False
    response_data["response_time_ms"] = round((time.perf_counter() - start) * 1000, 2)

    set_cache(cache_key_for_order(order_id), response_data)
    invalidate_list_cache()
    return response_data


@app.post("/orders/delete", status_code=status.HTTP_200_OK)
def delete_order(order_id: str, db: Session = Depends(get_db)):
    order = db.query(models.Order).filter(models.Order.order_id == order_id).first()
    if not order:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Order not found")

    db.delete(order)
    db.commit()

    delete_cache(cache_key_for_order(order_id))
    invalidate_list_cache()
    return {"detail": f"Order {order_id} deleted"}

import json
import logging
import redis

logger = logging.getLogger("redis_client")

REDIS_HOST = "localhost"
REDIS_PORT = 6379
REDIS_DB = 0
CACHE_TTL_SECONDS = 300

redis_client = redis.Redis(
    host=REDIS_HOST,
    port=REDIS_PORT,
    db=REDIS_DB,
    decode_responses=True,
    socket_connect_timeout=2,
    socket_timeout=2,
)


def cache_key_for_order(order_id: str) -> str:
    return f"order:{order_id}"


def cache_key_for_list() -> str:
    return "orders:all"


def get_cache(key: str):
    try:
        val = redis_client.get(key)
        return json.loads(val) if val is not None else None
    except redis.exceptions.RedisError as e:
        logger.warning(f"Redis unavailable during get_cache('{key}'): {e}")
        return None


def set_cache(key: str, value, ttl: int = CACHE_TTL_SECONDS):
    try:
        redis_client.set(key, json.dumps(value, default=str), ex=ttl)
    except redis.exceptions.RedisError as e:
        logger.warning(f"Redis unavailable during set_cache('{key}'): {e}")


def delete_cache(key: str):
    try:
        redis_client.delete(key)
    except redis.exceptions.RedisError as e:
        logger.warning(f"Redis unavailable during delete_cache('{key}'): {e}")


def invalidate_list_cache():
    try:
        redis_client.delete(cache_key_for_list())
    except redis.exceptions.RedisError as e:
        logger.warning(f"Redis unavailable during invalidate_list_cache(): {e}")

from pydantic import BaseModel, ConfigDict
from typing import Optional


class OrderBase(BaseModel):
    order_id: str
    customer_id: str
    customer_name: str
    customer_email: str
    product_id: str
    product_name: str
    category: Optional[str] = None
    sub_category: Optional[str] = None
    brand: Optional[str] = None
    unit_price: Optional[float] = None
    quantity: Optional[int] = None
    discount_percent: Optional[float] = None
    total_price: Optional[float] = None
    order_date: Optional[str] = None
    delivery_date: Optional[str] = None
    order_status: Optional[str] = None
    payment_method: Optional[str] = None
    payment_status: Optional[str] = None
    shipping_address: Optional[str] = None
    city: Optional[str] = None
    state: Optional[str] = None
    country: Optional[str] = None
    warehouse_id: Optional[str] = None


class OrderCreate(OrderBase):
    """Body for POST /orders/create"""
    pass


class OrderFetchRequest(BaseModel):
    """Body for POST /orders/fetch. Omit order_id to fetch all orders."""
    order_id: Optional[str] = None


class OrderResponse(OrderBase):
    """Returned to the client — includes cache metadata."""
    model_config = ConfigDict(from_attributes=True)
    cache_hit: bool = False
    response_time_ms: float = 0.0

import os
from openpyxl import load_workbook

from database import SessionLocal, engine, Base
import models

EXCEL_PATH = os.path.join(os.path.dirname(os.path.abspath(__file__)), "redis_learning_data.xlsx")
SHEET_NAME = "Order_Data"

COLUMNS = [
    "order_id", "customer_id", "customer_name", "customer_email",
    "product_id", "product_name", "category", "sub_category", "brand",
    "unit_price", "quantity", "discount_percent", "total_price",
    "order_date", "delivery_date", "order_status", "payment_method",
    "payment_status", "shipping_address", "city", "state", "country",
    "warehouse_id",
]

OPTIONAL_EXTRA_COLUMNS = {"cache_hit", "response_time_ms"}


def _to_float(value):
    try:
        return float(value) if value not in (None, "") else None
    except (TypeError, ValueError):
        return None


def _to_int(value):
    try:
        return int(value) if value not in (None, "") else None
    except (TypeError, ValueError):
        return None


def _row_to_order_kwargs(row_values):
    data = dict(zip(COLUMNS, row_values))
    data["unit_price"] = _to_float(data.get("unit_price"))
    data["quantity"] = _to_int(data.get("quantity"))
    data["discount_percent"] = _to_float(data.get("discount_percent"))
    data["total_price"] = _to_float(data.get("total_price"))

    if data["total_price"] is None and data["unit_price"] is not None and data["quantity"] is not None:
        discount = data["discount_percent"] or 0
        data["total_price"] = round(data["unit_price"] * data["quantity"] * (1 - discount), 2)

    for key in ("order_date", "delivery_date"):
        if data.get(key) is not None:
            data[key] = str(data[key])

    return data


def validate_excel_headers(excel_source, sheet_name: str = SHEET_NAME):
    try:
        wb = load_workbook(excel_source, data_only=True, read_only=True)
    except Exception as e:
        return False, f"File is not a valid Excel workbook: {e}"

    if sheet_name not in wb.sheetnames:
        available = ", ".join(wb.sheetnames)
        wb.close()
        return False, f"Sheet '{sheet_name}' not found. Available sheets: {available}"

    ws = wb[sheet_name]
    header_row = next(ws.iter_rows(min_row=1, max_row=1, values_only=True), None)
    wb.close()

    if not header_row:
        return False, "Header row is empty or missing."

    actual_headers = [str(h).strip() if h is not None else "" for h in header_row]
    missing = [col for col in COLUMNS if col not in actual_headers]
    if missing:
        return False, f"Missing required column(s): {', '.join(missing)}"

    unexpected = [h for h in actual_headers if h and h not in COLUMNS and h not in OPTIONAL_EXTRA_COLUMNS]
    if unexpected:
        return False, f"Unrecognized column(s) found: {', '.join(unexpected)}"

    return True, "Column headers are valid."


def seed_from_excel(db, excel_path=EXCEL_PATH, sheet_name: str = SHEET_NAME) -> dict:
    wb = load_workbook(excel_path, data_only=True, read_only=True)
    ws = wb[sheet_name]

    existing_ids = {oid for (oid,) in db.query(models.Order.order_id).all()}

    inserted, skipped, total_rows = 0, 0, 0
    rows = ws.iter_rows(min_row=2, values_only=True)

    for row in rows:
        if row is None or all(v is None for v in row):
            continue
        total_rows += 1

        kwargs = _row_to_order_kwargs(row[: len(COLUMNS)])
        order_id = kwargs.get("order_id")
        if not order_id:
            continue
        if order_id in existing_ids:
            skipped += 1
            continue

        db.add(models.Order(**kwargs))
        existing_ids.add(order_id)
        inserted += 1

    db.commit()
    wb.close()
    return {"inserted": inserted, "skipped_existing": skipped, "total_rows": total_rows}


if __name__ == "__main__":
    Base.metadata.create_all(engine)
    session = SessionLocal()
    try:
        summary = seed_from_excel(session)
        print(f"✅ Seed complete: {summary}")
    finally:
        session.close()
